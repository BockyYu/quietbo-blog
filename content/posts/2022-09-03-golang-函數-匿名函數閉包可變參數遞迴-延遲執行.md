## <span class="ez-toc-section" id="%E5%87%BD%E6%95%B8"></span>函數<span class="ez-toc-section-end"></span> 

輸入的參數可多個, 返回也可多個

<pre class="wp-block-code"><code class="">func 函數名(參數)(返回值){
    函數體
}</code></pre>

範例:

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import "fmt"

func main() {

    fmt.Println(sum(2, 3)) //5
    x, y := getXYValue()
    fmt.Printf("x:%d y:%d", x, y) //x:1 y:2

    var f func() //指定function類型
    f = process  //把process這個func指定給f
    f()          // 調用f等於調用process
}

func sum(x, y int) int {
    return x + y
}

func getXYValue() (x, y int) {
    x = 1
    y = 2
    return x, y
}

func process() {
    fmt.Println("use process function")
}</code></pre>

## <span class="ez-toc-section" id="%E5%8C%BF%E5%90%8D%E5%87%BD%E6%95%B8"></span>匿名函數<span class="ez-toc-section-end"></span> 

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import "fmt"

func main() {
    //匿名函數

    f := func(name string) {
        fmt.Println("hello,", name)
    }
    f("bocky") //hello, bocky
    fmt.Println("===")

    iterator([]int{1, 2, 3, 4, 5, 6}, func(value int) { //函數沒有名稱
        fmt.Println(value)
    })
}

func iterator(arr []int, ff func(int)) { // 這裡的函數是有名稱ff
    for _, value := range arr {
        ff(value)
    }
}</code></pre>

## <span class="ez-toc-section" id="%E9%96%89%E5%8C%85"></span>閉包<span class="ez-toc-section-end"></span> 

閉包 = 函數+引用環境

範例:

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import "fmt"

func main() {
    //str := "Bocky" // IDE會出現str declared but not used
    f := func() {
        str := "Abby"
        fmt.Println(str) //Abby
    }
    f()

    //  閉包擁有記憶性
    fmt.Println("======")
    add1 := add(1)
    fmt.Println(add1())       //2
    fmt.Println(add1())       //3
    fmt.Printf("%p\n", &add1) //0xc000006030

    fmt.Println("======")
    add2 := add(200)
    fmt.Println(add2())       //201
    fmt.Println(add2())       //202
    fmt.Printf("%p\n", &add2) //0xc000006038

    fmt.Println("======")
    bocky := user("bocky")
    name, hp := bocky()
    fmt.Println(name, hp) //bocky 1000

    fmt.Println("======")
    amy := user("amy")
    name2, hp2 := amy()
    fmt.Println(name2, hp2) //amy 1000
}

func add(v1 int) func() int {
    return func() int {
        v1++
        return v1
    }
}

func user(name string) func() (string, int) {
    hp := 1000
    return func() (string, int) {
        return name, hp
    }
}</code></pre>

## <span class="ez-toc-section" id="%E5%8F%AF%E8%AE%8A%E5%8F%83%E6%95%B8"></span>可變參數<span class="ez-toc-section-end"></span> 

在實際開發中，總有一些函數的參數個數是在編碼過程中無法確定的，  
當有狀況是需要實現類似的接口時，就會用到&#8221;可變參數&#8221;了。

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import (
    "bytes"
    "fmt"
)

func main() {
    //可變參數
    funx1(1)
    funx1(1, 2, 3, 5)

    funx2(11)                   //11 int
    funx2("1234")               //1234 string
    funx2(1.23)                 //1.23 default
    funx2(func() () { return }) //0x96ca00 default

    fmt.Println("=======")
    fmt.Println(funx3("hello "))
    fmt.Println(funx3("hello ", "I'm "))           //hello I'm
    fmt.Println(funx3("hello ", "I'm ", "Bocky!")) //hello I'm Bocky!



    fmt.Println("=======")
    Tprint("A","B","C")
}

func funx1(args ...int) { //不確定會傳多少個int的時候可以使用...但一定要是int類型
    for _, arg := range args {
        fmt.Println(arg)
    }
}

func funx2(args ...interface{}) { //任一類型的可變參數
    for _, arg := range args {
        switch arg.(type) {
        case int:
            fmt.Println(arg, "int")
        case string:
            fmt.Println(arg, "string")
        case float32:
            fmt.Println(arg, "float32")
        default:
            fmt.Println(arg, "default")
        }
    }
}

func funx3(strs ...string) string {
    var b bytes.Buffer
    for _, s := range strs { // 依序遍歷strs內的string
        b.WriteString(s)
    }
    return b.String()
}

//下方Tprint會call funx4來執行print的動作
func Tprint(params ...interface{}) {
    funx4(params)
}
func funx4(params ...interface{}) {
    for _, value := range params {
        fmt.Println(value)
    }
}</code></pre>

## <span class="ez-toc-section" id="%E8%A8%88%E7%AE%97%E5%87%BD%E6%95%B8%E6%99%82%E9%96%93"></span>計算函數時間<span class="ez-toc-section-end"></span> 

想要獲取計算某個任務從開始到結束的運行時間，我們可以使用time包中的Slice函數，用法:

<pre class="wp-block-code"><code class="">func Since (t Time) Duration</code></pre>

t是時間值。  
**此方法是time.Now().Sub(t)的快捷方式**

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import (
    "fmt"
    "time"
)

func main() {
    //func Since(t Time)Duration
    //time.Now().sub(t
    testTime()
}

func testTime() {

    start := time.Now()
    sum := 0
    for i := 0; i &lt; 1000000000; i++ {
        sum++
    }
    duration := time.Since(start)
    fmt.Println("function執行時間:", duration)
}</code></pre>

## <span class="ez-toc-section" id="%E9%81%9E%E8%BF%B4"></span>遞迴<span class="ez-toc-section-end"></span> 

費波那契數列

<pre class="wp-block-code"><code class="">1,1,2,3,5,8,13 ,...</code></pre>

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import "fmt"

func main() {
    //遞迴

    result := 0
    for i := 1; i &lt;= 10; i++ {
        result = fibonacci(i)
        fmt.Printf("fibonacci(%2d) -&gt; %3d\n", i, result)
    }
}

func fibonacci(n int) (rtn int) {
    if n &lt;= 2 {
        rtn = 1
    } else {
        rtn = fibonacci(n-1) + fibonacci(n-2)
    }
    return
}</code></pre>

## <span class="ez-toc-section" id="defer_%E5%BB%B6%E9%81%B2%E5%9F%B7%E8%A1%8C"></span>defer 延遲執行<span class="ez-toc-section-end"></span> 

<ul class="wp-block-list">
  <li>
    用於資源的釋放，會在函數返回之前進行調用，簡單來說就是宣告 function &#8220;結束前&#8221;的動作。
  </li>
  <li>
    先進後出
  </li>
  <li>
    會保留當下狀態
  </li>
</ul>

簡單的範例:

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import (
    "fmt"
)


func main() {
    fmt.Println("defer start")
    defer fmt.Println(1)
    defer fmt.Println(2)
    defer fmt.Println(3)
    fmt.Println("defer stop")
}</code></pre>

結果:

<pre class="wp-block-code"><code class="">defer start
defer stop
3
2
1</code></pre>

<pre class="wp-block-code"><code lang="go" class="language-go line-numbers">package main

import (
    "fmt"
)

func main() {
    num := 1
    x := addOne(num)

    fmt.Printf("Result is %d\n",x)
}


func closeProcess() {
    fmt.Println("close process")
}
func addOne(x int) int {
    fmt.Println("Start process")
    defer closeProcess() // function結束前返回
    if x &gt; 2 {
        fmt.Println("x &gt; 2")
        return x * 20
    } else {
        fmt.Println("x &lt;=2")
        return x + 1
    }
}</code></pre>

結果:

<pre class="wp-block-code"><code class="">Start process
x &lt;=2
close process
Result is 2</code></pre>
